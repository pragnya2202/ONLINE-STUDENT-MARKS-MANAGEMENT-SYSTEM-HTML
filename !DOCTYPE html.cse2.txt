<script>

let students = []

function showSection(id){

let sections=document.querySelectorAll('.section')

sections.forEach(section=>section.classList.remove('active'))

document.getElementById(id).classList.add('active')

}

function login(){

let role=document.getElementById("role").value

if(role==="teacher")
showSection("teacher")

else if(role==="student")
showSection("student")

else
alert("Please select role")

}

function saveMarks(){

let name=document.getElementById("studentName").value
let math=document.getElementById("math").value
let science=document.getElementById("science").value
let english=document.getElementById("english").value

let total = Number(math)+Number(science)+Number(english)

let grade=""

if(total>=270) grade="A"
else if(total>=240) grade="B"
else if(total>=200) grade="C"
else grade="D"

students.push({name,math,science,english,total,grade})

alert("Marks Saved")

}

function viewMarks(){

let name=document.getElementById("searchStudent").value

let student=students.find(s=>s.name===name)

if(student){

document.getElementById("marksResult").innerHTML=`

<h5>Marks</h5>

Math : ${student.math}<br>
Science : ${student.science}<br>
English : ${student.english}<br>

Total : ${student.total}<br>
Grade : ${student.grade}

`

}

else{
document.getElementById("marksResult").innerHTML="No Record Found"
}

}

function generateReport(){

let reportHTML=""

students.forEach(s=>{

reportHTML+=`

<div class="border p-3 mb-3">

<h5>${s.name}</h5>

Math : ${s.math} <br>
Science : ${s.science} <br>
English : ${s.english} <br>

<b>Total : ${s.total}</b><br>
<b>Grade : ${s.grade}</b>

</div>

`

})

document.getElementById("reportCard").innerHTML=reportHTML

}
